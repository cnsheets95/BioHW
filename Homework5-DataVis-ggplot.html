Create 4 plots:

  • box plots
  • raw data + point and error bars (mean and 95% conf. int.)

All these plots should have the same general layout but change the type of plot to visualize the data.

HINT: logbodysize and trophic_position need to be turned into long format

logBloom_df <- bloom_df %>%
  group_by(reg) %>%
  summarise(mean = mean(logbodysize, na.rm = TRUE), #log mean
            sd = sd(logbodysize, na.rm = TRUE), #log standard deviation
            n = n()) %>% #log count
  mutate(se = sd / sqrt(n), #log standard error
         ci = 1.96 * se) #log 95% confidence interval
## `summarise()` ungrouping output (override with `.groups` argument)
logBloom_df$Measurement <- 'logbodysize'

trophicBloom_df <- bloom_df %>%
  group_by(reg) %>%
  summarise(mean = mean(trophic_position, na.rm = TRUE), #troph mean
            sd = sd(trophic_position, na.rm = TRUE), #troph standard deviation
            n = n()) %>% #troph count
  mutate(se = sd / sqrt(n), #troph standard error
         ci = 1.96*se) #troph 95% confidence interval
## `summarise()` ungrouping output (override with `.groups` argument)
trophicBloom_df$Measurement <- 'trophic_position'

bloom_sum_df <- rbind(logBloom_df,trophicBloom_df)
